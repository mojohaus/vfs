22  import java.io.File;
23  
24  import org.apache.commons.vfs2.FileObject;
25  import org.apache.commons.vfs2.VFS;
26  import org.codehaus.mojo.vfs.internal.DefaultMergeVfsMavenRepositories;
27  import org.codehaus.mojo.vfs.internal.DefaultVfsFileSetManager;
28  import org.codehaus.plexus.util.FileUtils;
29  import org.junit.Assert;
30  import org.junit.Test;
31  
32  public class MavenReposMergeTest
33      extends AbstractVfsTestCase
34  {
35      private VfsFileSetManager filesetManager = new DefaultVfsFileSetManager();
36  
37      private MergeVfsMavenRepositories repoMerger = new DefaultMergeVfsMavenRepositories();
38  
39      public void runTestMavenReposMerge( boolean dryRun )
40          throws Exception
41      {
42  
43          basedir = basedir.getAbsoluteFile();
44          builddir = builddir.getAbsoluteFile();
45  
46          File testDirectory = new File( builddir, "test-repos" );
47          FileUtils.deleteDirectory( testDirectory );
48  
49          FileObject scmSourceRepo =
50              VFS.getManager().resolveFile( "file://" + new File( basedir, "src/test/data/repos/source" ) );
51  
52          FileObject scmTargetRepo =
53              VFS.getManager().resolveFile( "file://" + new File( basedir, "src/test/data/repos/target" ) );
54  
55          FileObject testSourceRepo =
56              VFS.getManager().resolveFile( "file://" + new File( builddir, "test-repos/source" ) );
57  
58          FileObject testTargetRepo =
59              VFS.getManager().resolveFile( "file://" + new File( builddir, "test-repos/target" ) );
60  
61          VfsFileSet fileset = new VfsFileSet();
62          String[] allFiles = { "**" };
63          fileset.setIncludes( allFiles );
64          String[] hiddenFiles = { "**/.*/**" }; // dont want local svn metadata file sneaking in
65          fileset.setExcludes( hiddenFiles );
66  
67          // prep source repo
68          fileset.setSource( scmSourceRepo );
69          fileset.setDestination( testSourceRepo );
70          filesetManager.copy( fileset );
71  
72          // prep target repo
73          fileset.setSource( scmTargetRepo );
74          fileset.setDestination( testTargetRepo );
75          filesetManager.copy( fileset );
76  
77          File stagingDir = new File( builddir, "test-repos/staging" );
78  
79          repoMerger.merge( testSourceRepo, testTargetRepo, stagingDir, dryRun );
80  
81      }
82  
83      @Test
84      public void testMavenReposMerge()
85          throws Exception
86      {
87          this.runTestMavenReposMerge( false );
88  
89          File targetDir = new File( builddir, "test-repos/target" );
90  
91          Assert.assertTrue( new File( targetDir, "com/company/package/artifact1/3.0/artifact1-3.0.pom" ).exists() );
92          Assert.assertTrue( new File( targetDir, "com/company/package/artifact1/maven-metadata.xml" ).exists() );
93          Assert.assertTrue( new File( targetDir, "com/company/package/artifact1/maven-metadata.xml.md5" ).exists() );
94          Assert.assertTrue( new File( targetDir, "com/company/package/artifact1/maven-metadata.xml.sha1" ).exists() );
95  
96          Assert.assertTrue( new File( targetDir, "com/company/package/artifact3/1.0/artifact3-1.0.pom" ).exists() );
97          Assert.assertTrue( new File( targetDir, "com/company/package/artifact3/maven-metadata.xml" ).exists() );
98  
99          // should not see hashfile since it is a brand new artifacts
100         Assert.assertFalse( new File( targetDir, "com/company/package/artifact3/maven-metadata.xml.md5" ).exists() );
101         Assert.assertFalse( new File( targetDir, "com/company/package/artifact3/maven-metadata.xml.sha1" ).exists() );
102 
103         File stagingDir = new File( builddir, "test-repos/staging" );
104         Assert.assertFalse( stagingDir.exists() );
105 
106     }
107 
108     @Test
109     public void testMavenReposMergeWithDryrun()
110         throws Exception
111     {
112         this.runTestMavenReposMerge( true );
113 
114         File stagingDir = new File( builddir, "test-repos/staging" );
115         Assert.assertTrue( stagingDir.exists() );
116 
117     }
